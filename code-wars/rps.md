

Rock Paper Scissors!
https://www.codewars.com/kata/5672a98bdbdd995fad00000f/javascript

```js
const rps = (p1, p2) => {
  if (p1 === p2) return "Draw!";
  var rules = {rock: "scissors", paper: "rock", scissors: "paper"};
  if (p2 === rules[p1]) {
    return "Player 1 won!";
  }
  else {
    return "Player 2 won!";
  }
};
```
